import os

@dp.message(CommandStart())
async def start(message: types.Message):
    await message.answer(WELCOME_TEXT)


@dp.message()
async def recv_user_message(message: types.Message):
    if not message.text:
        return await message.answer("Iltimos, faqat matnli eâ€™tiroz yuboring.")

    sent = await bot.send_message(
        GROUP_ID,
        f"ðŸ“© Yangi murojaat:\n\n{message.text}"
    )

    message_links[sent.message_id] = message.from_user.id

    await message.answer("Murojaatingiz qabul qilindi! âœ”")


@dp.message()
async def group_reply(message: types.Message):
    if message.chat.id == GROUP_ID and message.reply_to_message:
        replied_msg_id = message.reply_to_message.message_id

        if replied_msg_id in message_links:
            user_id = message_links[replied_msg_id]

            await bot.send_message(
                user_id,
                f"ðŸ’¬ Javob:\n\n{message.text}"
            )


async def main():
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
